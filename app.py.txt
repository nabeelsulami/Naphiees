import streamlit as st
import pandas as pd
import os

# 1. إعدادات الصفحة والتصميم
st.set_page_config(page_title="محرك نفيس الذكي", page_icon="🔍", layout="wide")

DB_FILE = "nphies_data.xlsx"

# 2. تأسيس قاعدة البيانات الافتراضية لو لم تكن موجودة
def init_database():
    if not os.path.exists(DB_FILE):
        data = {
            'Nphies_Code': ['83600-00-20', '83600-00-10', '83600-00-15'],
            'Service_Name_EN': ['Consultant Consultation', 'Specialist Consultation', 'Registrar Consultation'],
            'Service_Name_AR': ['كشفية طبيب استشاري', 'كشفية طبيب أخصائي', 'كشفية طبيب نائب / أخصائي أول'],
            'Keywords_Descriptions': [
                'استشاري، عيادة، كشف، معاينة، فحص، تقييم حالة، دكتور، مراجعة، Consultant, Clinic, Exam, Visit',
                'أخصائي، عيادة، كشف، فحص، Specialist, Clinic, Exam',
                'نائب، أخصائي أول، كشف، عيادة، Registrar, Senior Specialist'
            ]
        }
        df = pd.DataFrame(data)
        df.to_excel(DB_FILE, index=False)

init_database()

# 3. دالة قراءة البيانات الحية
@st.cache_data(ttl=2)
def load_data():
    if os.path.exists(DB_FILE):
        try:
            df = pd.read_excel(DB_FILE)
            required_cols = ['Nphies_Code', 'Service_Name_EN', 'Service_Name_AR', 'Keywords_Descriptions']
            for col in required_cols:
                if col not in df.columns:
                    df[col] = ""
            for col in df.columns:
                df[col] = df[col].astype(str).str.strip().replace('nan', '')
            return df[required_cols]
        except:
            return pd.DataFrame()
    return pd.DataFrame()

df = load_data()

# 4. واجهة البحث للموظفين والمستخدمين
st.markdown("""
    <style>
    .main-title { font-size: 32px; font-weight: bold; text-align: center; color: #1E3A8A; margin-bottom: 5px; }
    .sub-title { font-size: 16px; text-align: center; color: #4B5563; margin-bottom: 30px; }
    .footer { text-align: center; margin-top: 50px; font-size: 14px; color: #1E3A8A; font-weight: bold; border-top: 1px solid #E5E7EB; padding-top: 20px; }
    </style>
    """, unsafe_allow_html=True)

st.markdown('<div class="main-title">🔍 محرك البحث والربط الذكي لأكواد نفيس</div>', unsafe_allow_html=True)
st.markdown('<div class="sub-title">نظام الاستعلام الموحد للخدمات والمترادفات الطبية الميدانية</div>', unsafe_allow_html=True)

# شريط البحث الرئيسي المتاح للجميع
search_query = st.text_input("✍️ ابحث هنا بكتابة الكود، اسم الخدمة (عربي/إنجليزي)، أو الكلمات التوضيحية البديلة:", "").strip()

if search_query and not df.empty:
    filtered_df = df[
        df['Nphies_Code'].str.contains(search_query, case=False, na=False) |
        df['Service_Name_EN'].str.contains(search_query, case=False, na=False) |
        df['Service_Name_AR'].str.contains(search_query, case=False, na=False) |
        df['Keywords_Descriptions'].str.contains(search_query, case=False, na=False)
    ]
else:
    filtered_df = df

st.markdown(f"##### 📊 نتائج البحث ({len(filtered_df)} خدمة مطابقة):")

if not filtered_df.empty:
    display_df = filtered_df.rename(columns={
        'Nphies_Code': 'كود نفيس الموحد',
        'Service_Name_EN': 'اسم الخدمة (EN)',
        'Service_Name_AR': 'اسم الخدمة (AR)',
        'Keywords_Descriptions': 'الجمل التوضيحية والكلمات الدلالية'
    })
    st.dataframe(display_df, use_container_width=True, hide_index=True)
else:
    st.warning("⚠️ لم يتم العثور على أي خدمة مطابقة للبحث.")

st.sidebar.markdown("### 🔒 بوابة الإدارة والأمان")
show_admin = st.sidebar.checkbox("فتح لوحة التحكم والتحديث")

if show_admin:
    password = st.sidebar.text_input("أدخل كلمة المرور السرية:", type="password")
    
    if password == "admin123":
        st.sidebar.success("أهلاً بك دكتور نبيل! اختر طريقة تحديث البيانات:")
        
        admin_option = st.sidebar.radio("آلية التحديث المطلوبة:", ["إدخال يدوي (كود فردي)", "رفع ملف Excel (تحديث جماعي)"])
        
        if admin_option == "إدخال يدوي (كود فردي)":
            with st.sidebar.form(key="manual_form", clear_on_submit=True):
                new_code = st.text_input("كود نفيس الموحد:")
                new_name_en = st.text_input("اسم الخدمة (EN):")
                new_name_ar = st.text_input("اسم الخدمة (AR):")
                new_keywords = st.text_area("الكلمات الدلالية والمترادفات (الجمل التوضيحية):")
                
                submit_manual = st.form_submit_button("➕ حفظ الكود يدوياً")
                
                if submit_manual:
                    if new_code and new_name_en:
                        current_df = pd.read_excel(DB_FILE)
                        new_row = {
                            'Nphies_Code': new_code, 
                            'Service_Name_EN': new_name_en, 
                            'Service_Name_AR': new_name_ar, 
                            'Keywords_Descriptions': new_keywords
                        }
                        updated_df = pd.concat([current_df, pd.DataFrame([new_row])], ignore_index=True)
                        updated_df.to_excel(DB_FILE, index=False)
                        st.sidebar.success(f"✅ تم إضافة الكود {new_code} بنجاح!")
                        st.rerun()
                    else:
                        st.sidebar.error("⚠️ يجب كتابة الكود والاسم الإنجليزي كحد أدنى.")
                        
        elif admin_option == "رفع ملف Excel (تحديث جماعي)":
            st.sidebar.markdown("""
            💡 **تنبيه لهيكل ملف الـ Excel المرفوع:**
            `Nphies_Code`, `Service_Name_EN`, `Service_Name_AR`, `Keywords_Descriptions`
            """)
            uploaded_excel = st.sidebar.file_uploader("📁 اختر ملف الـ Excel لدمجه مع النظام:", type=["xlsx", "xls"])
            
            if uploaded_excel is not None:
                try:
                    uploaded_df = pd.read_excel(uploaded_excel)
                    if 'Nphies_Code' in uploaded_df.columns and 'Service_Name_EN' in uploaded_df.columns:
                        current_df = pd.read_excel(DB_FILE)
                        final_df = pd.concat([current_df, uploaded_df], ignore_index=True).drop_duplicates(subset=['Nphies_Code'], keep='last')
                        final_df.to_excel(DB_FILE, index=False)
                        st.sidebar.success("✅ تم دمج وتحديث قاعدة البيانات من ملف الـ Excel بنجاح!")
                        st.rerun()
                    else:
                        st.sidebar.error("❌ خطأ: أسماء الأعمدة غير مطابقة!")
                except Exception as e:
                    st.sidebar.error(f"حدث خطأ أثناء معالجة الملف: {e}")
                    
    elif password != "":
        st.sidebar.error("❌ كلمة المرور خاطئة!")
